 23class Icon:
 24    """
 25    Icon for `OutputItem`s
 26
 27    If unset, defaults to the workflow icon.
 28    """
 29
 30    path: str = ""
 31    """
 32    A resource to load an item from.
 33
 34    Can be a path to an image file (if `type` unset), a path to a file whose icon is to be used (if `type` is
 35    `fileicon`, or a UTI (if `type` is `filetype`).
 36    """
 37    type: Optional[str] = None
 38    """The type of resource that `path` specifies. Must be one of ``, `fileicon`, `filetype`."""
 39
 40    @classmethod
 41    def image(cls, path: str) -> "Icon":
 42        """
 43        Get an icon with the contents of the image at `path`
 44
 45        :return: an `Icon` that can be used in `OutputItem`
 46        """
 47        return Icon(path)
 48
 49    @classmethod
 50    def file_icon(cls, path: str) -> "Icon":
 51        """
 52        Get an icon with the same icon as the file at `path`
 53
 54        For example, to get a calendar icon, specify the path of the macOS built-in Calendar app:
 55
 56        ```python
 57        Icon.file_icon(path="/System/Applications/Calendar.app")
 58        ```
 59
 60        :return: an `Icon` that can be used in `OutputItem`
 61        """
 62        return Icon(path, type="fileicon")
 63
 64    @classmethod
 65    def uti(cls, uti: str) -> "Icon":
 66        """
 67        Get an icon with the same icon as the file at `path`
 68
 69        macOS comes with built-in icons for many types of data, declared as Uniform Type Identifiers (UTIs).
 70        We can tell Alfred to use these icons by just specifying the UTI.
 71
 72        An outdated list of available UTIs can be found on the [Apple website](https://developer.apple.com/library/archive/documentation/Miscellaneous/Reference/UTIRef/Articles/System-DeclaredUniformTypeIdentifiers.html#//apple_ref/doc/uid/TP40009259-SW1).
 73
 74        It's possible to dump the available UTIs on a system using the following command:
 75        ```sh
 76        /System/Library/Frameworks/CoreServices.framework/Frameworks/LaunchServices.framework/Versions/A/Support/lsregister -dump | grep uti: | cut -c 29- | sort | uniq
 77        ```
 78
 79        To use the macOS icon for a JPEG image:
 80        ```python
 81        Icon.uti(uti="public.jpeg")
 82        ```
 83
 84        :return: an `Icon` that can be used in `OutputItem`
 85        """  # noqa: E501
 86        return Icon(uti, type="filetype")

 89        if self.type and self.type not in ("fileicon", "filetype"):
 90            raise ValueError("if set, type must be either fileicon or filetype")

 93class Type(Enum):
 94    """
 95    The type of output
 96
 97    By specifying `File`, Alfred treats your result as a file on your system. This allows the user to perform actions on
 98    the file like they can with Alfred's standard file filters.
 99
100    When returning files, Alfred will check if they exist before presenting that result to the user. This has a very
101    small performance implication but makes results as predictable as possible. If you would like Alfred to skip this
102    check because you are certain the files you are returning exist, use `FileSkipCheck`.
103    """
104
105    Default = "default"
106    File = "file"
107    FileSkipCheck = "file:skipcheck"

240class Environment:
241    """
242    The environment variables passed by Alfred
243
244    See [here](https://www.alfredapp.com/help/workflows/script-environment-variables/) for more details.
245    """
246
247    debug: bool
248    """Whether Alfred is running in debug mode"""
249    preferences_file: Path
250    """The path to Alfred's preferences file"""
251    version: str
252    """The Alfred version (e.g., `v5.0.0`)"""
253    version_build: str
254    """The Alfred build (e.g., `1234`)"""
255    workflow_name: str
256    """The name of the workflow being executed"""
257    workflow_version: Optional[str]
258    """The version of the workflow"""
259    workflow_bundle_id: Optional[str]
260    """The bundle ID of the workflow, if set"""
261    workflow_uid: str
262    """The UID of the workflow"""
263    workflow_cache: Optional[Path]
264    """The path to the workflow's cache directory"""
265    workflow_data: Optional[Path]
266    """The path to the workflow's data directory"""
267
268    @classmethod
269    def from_env(cls) -> Optional["Environment"]:
270        """
271        Get Alfred's environment variables
272
273        :return: a model populated from the environment variables set by Alfred
274        """
275
276        if not os.environ.get("alfred_version"):
277            logging.warning("Not running in an Alfred environment")
278            return None
279
280        cache_dir = os.environ.get("alfred_workflow_cache")
281        data_dir = os.environ.get("alfred_workflow_data")
282        cast(str, os.environ.get("alfred_preferences"))
283
284        return Environment(
285            debug=(os.environ.get("alfred_debug") == "1"),
286            preferences_file=Path(cast(str, os.environ.get("alfred_preferences"))),
287            version=cast(str, os.environ.get("alfred_version")),
288            version_build=cast(str, os.environ.get("alfred_version_build")),
289            workflow_name=cast(str, os.environ.get("alfred_workflow_name")),
290            workflow_version=os.environ.get("alfred_workflow_version"),
291            workflow_bundle_id=os.environ.get("alfred_workflow_bundle_id"),
292            workflow_uid=os.environ.get("alfred_workflow_uid") or "",
293            workflow_cache=Path(cache_dir).expanduser() if cache_dir else None,
294            workflow_data=Path(data_dir).expanduser() if data_dir else None,
295        )
296
297    @property
298    def preferences(self) -> dict:
299        """
300        Get Alfred's preferences
301
302        :return: a dictionary representation of the Alfred preferences file
303        """
304        with self.preferences_file.open("rb") as f:
305            prefs = plistlib.load(f)
306        return prefs
